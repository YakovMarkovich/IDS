### Quick start

install <u>**Imagine**</u> package into your application

```bash

cd yourApp

npm install --save Imagine
```

### Start using <u>`Divider` </u> component

<br />
<br />

##### Example:

```javascript
import { ImagineProvider, Container, Divider, Text } from "imagine";

function App() {
  return (
    <ImagineProvider>
      <Container
        flexDirection={"col"}
        justifyContent={"center"}
        alignItems={"center"}
      >
        <Text text={"טקסט מעל קו המפריד"} />
        <Divider />
        <Text text={"טקסט מתחת קו המפריד"} />
      </Container>
      <Container
        flexDirection={"row"}
        justifyContent={"center"}
        alignItems={"center"}
      >
        <Text text={"טקסט בצד ימין"} />
        <Divider type='vertical'/>
        <Text text={"טקסט מצד שמאל"} />
      </Container>
    </ImagineProvider>
  );
}
```
